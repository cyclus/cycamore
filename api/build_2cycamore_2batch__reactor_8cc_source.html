00710   using cyclus::Material;
00711 
00712   std::set<RequestPortfolio<Material>::Ptr> set;
00713   double order_size;
00714 
00715   switch (phase()) {
00716     // the initial phase requests as much fuel as necessary to achieve an entire
00717     // core
00718     case INITIAL:

00721                    - spillover_->quantity();

00951                                     << " is adding " << mat->quantity()
00952                                     << " of material to its reserves.";
00953 
00954   // this is a hack! Whatever *was* left in spillover now magically becomes this
00955   // new commodity. We need to do something different (maybe) for recycle.
00956   spillover_->Absorb(mat);
00957 
00958   while (!cyclus::IsNegative(spillover_->quantity() - batch_size())) {
00959     Material::Ptr batch;

00961     // the amount of fuel in spillover

00963       batch = spillover_->ExtractQty(batch_size());
00964     } else {
00965       batch = spillover_->ExtractQty(spillover_->quantity());
00966     }
00967     assert(commod != "");
00968     crctx_.AddRsrc(commod, batch);
00969     reserves_.Push(batch);
00970   }
