165 std::set<cyclus::BidPortfolio<cyclus::Material>::Ptr> Enrichment::GetMatlBids(
166  cyclus::CommodMap<cyclus::Material>::type& out_requests) {
167  using cyclus::Bid;
168  using cyclus::BidPortfolio;
169  using cyclus::CapacityConstraint;
170  using cyclus::Converter;
171  using cyclus::Material;
172  using cyclus::Request;
173  using cyclus::toolkit::MatVec;
174 
175  std::set<BidPortfolio<Material>::Ptr> ports;
176 
177  if ((out_requests.count(tails_commod) > 0) && (tails.quantity() > 0)) {
178  BidPortfolio<Material>::Ptr tails_port(new BidPortfolio<Material>());
179 
180  std::vector<Request<Material>*>& tails_requests =
181  out_requests[tails_commod];
182  std::vector<Request<Material>*>::iterator it;
183  for (it = tails_requests.begin(); it != tails_requests.end(); ++it) {
184  // offer bids for all tails material, keeping discrete quantities
185  // to preserve possible variation in composition
186  MatVec mats = tails.PopN(tails.count());
187  tails.Push(mats);
188  for (int k = 0; k < mats.size(); k++) {
189  Material::Ptr m = mats[k];
190  Request<Material>* req = *it;
191  tails_port->AddBid(req, m, this);
192  }
193  }
194  // overbidding (bidding on every offer)
195  // add an overall capacity constraint
196  CapacityConstraint<Material> tails_constraint(tails.quantity());
197  tails_port->AddConstraint(tails_constraint);
198  LOG(cyclus::LEV_INFO5, "EnrFac") << prototype()
199  << " adding tails capacity constraint of "
200  << tails.capacity();
201  ports.insert(tails_port);
202  }
203 
204  if ((out_requests.count(product_commod) > 0) && (inventory.quantity() > 0)) {
205  BidPortfolio<Material>::Ptr commod_port(new BidPortfolio<Material>());
206 
207  std::vector<Request<Material>*>& commod_requests =
208  out_requests[product_commod];
209  std::vector<Request<Material>*>::iterator it;
210  for (it = commod_requests.begin(); it != commod_requests.end(); ++it) {
211  Request<Material>* req = *it;
212  Material::Ptr mat = req->target();
213  double request_enrich = cyclus::toolkit::UraniumAssay(mat);
214  if (ValidReq(req->target()) &&
215  ((request_enrich < max_enrich) ||
216  (cyclus::AlmostEq(request_enrich, max_enrich)))) {
217  Material::Ptr offer = Offer_(req->target());
218  commod_port->AddBid(req, offer, this);
219  }
220  }
221 
222  Converter<Material>::Ptr sc(new SWUConverter(FeedAssay(), tails_assay));
223  Converter<Material>::Ptr nc(new NatUConverter(FeedAssay(), tails_assay));
224  CapacityConstraint<Material> swu(swu_capacity, sc);
225  CapacityConstraint<Material> natu(inventory.quantity(), nc);
226  commod_port->AddConstraint(swu);
227  commod_port->AddConstraint(natu);
228 
229  LOG(cyclus::LEV_INFO5, "EnrFac")
230  << prototype() << " adding a swu constraint of " << swu.capacity();
231  LOG(cyclus::LEV_INFO5, "EnrFac")
232  << prototype() << " adding a natu constraint of " << natu.capacity();
233  ports.insert(commod_port);
234  }
235  return ports;
236 }

292 void Enrichment::AddMat_(cyclus::Material::Ptr mat) {
293  // Elements and isotopes other than U-235, U-238 are sent directly to tails
294  cyclus::CompMap cm = mat->comp()->atom();
295  bool extra_u = false;
296  bool other_elem = false;
297  for (cyclus::CompMap::const_iterator it = cm.begin(); it != cm.end(); ++it) {
298  if (pyne::nucname::znum(it->first) == 92) {
299  if (pyne::nucname::anum(it->first) != 235 &&
300  pyne::nucname::anum(it->first) != 238 && it->second > 0) {
301  extra_u = true;
302  }
303  } else if (it->second > 0) {
304  other_elem = true;
305  }
306  }
307  if (extra_u) {
308  cyclus::Warn<cyclus::VALUE_WARNING>(
309  "More than 2 isotopes of U. "
310  "Istopes other than U-235, U-238 are sent directly to tails.");
311  }
312  if (other_elem) {
313  cyclus::Warn<cyclus::VALUE_WARNING>(
314  "Non-uranium elements are "
315  "sent directly to tails.");
316  }
317 
318  LOG(cyclus::LEV_INFO5, "EnrFac") << prototype() << " is initially holding "
319  << inventory.quantity() << " total.";
320 
321  try {
322  inventory.Push(mat);
323  } catch (cyclus::Error& e) {
324  e.msg(Agent::InformErrorMsg(e.msg()));
325  throw e;
326  }
327 
328  LOG(cyclus::LEV_INFO5, "EnrFac")
329  << prototype() << " added " << mat->quantity() << " of " << feed_commod
330  << " to its inventory, which is holding " << inventory.quantity()

350 // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
351 cyclus::Material::Ptr Enrichment::Enrich_(cyclus::Material::Ptr mat,
352  double qty) {
353  using cyclus::Material;
354  using cyclus::ResCast;
355  using cyclus::toolkit::Assays;
356  using cyclus::toolkit::UraniumAssay;
357  using cyclus::toolkit::SwuRequired;
358  using cyclus::toolkit::FeedQty;
359  using cyclus::toolkit::TailsQty;
360 
361  // get enrichment parameters
362  Assays assays(FeedAssay(), UraniumAssay(mat), tails_assay);
363  double swu_req = SwuRequired(qty, assays);
364  double natu_req = FeedQty(qty, assays);
365 
366  // Determine the composition of the natural uranium
367  // (ie. U-235+U-238/TotalMass)
368  double pop_qty = inventory.quantity();
369  Material::Ptr natu_matl = inventory.Pop(pop_qty, cyclus::eps_rsrc());
370  inventory.Push(natu_matl);
371 
372  cyclus::toolkit::MatQuery mq(natu_matl);
373  std::set<cyclus::Nuc> nucs;
374  nucs.insert(922350000);
375  nucs.insert(922380000);
376  double natu_frac = mq.mass_frac(nucs);
377  double feed_req = natu_req / natu_frac;
378 
379  // pop amount from inventory and blob it into one material
380  Material::Ptr r;
381  try {
382  // required so popping doesn't take out too much
383  if (cyclus::AlmostEq(feed_req, inventory.quantity())) {
384  r = cyclus::toolkit::Squash(inventory.PopN(inventory.count()));
385  } else {
386  r = inventory.Pop(feed_req, cyclus::eps_rsrc());
387  }
388  } catch (cyclus::Error& e) {
389  NatUConverter nc(FeedAssay(), tails_assay);
390  std::stringstream ss;
391  ss << " tried to remove " << feed_req << " from its inventory of size "
392  << inventory.quantity()
393  << " and the conversion of the material into natu is "
394  << nc.convert(mat);
395  throw cyclus::ValueError(Agent::InformErrorMsg(ss.str()));
396  }
397 
398  // "enrich" it, but pull out the composition and quantity we require from the
399  // blob
400  cyclus::Composition::Ptr comp = mat->comp();
401  Material::Ptr response = r->ExtractComp(qty, comp);
402  tails.Push(r);
403 
404  current_swu_capacity -= swu_req;
405 
406  intra_timestep_swu_ += swu_req;
407  intra_timestep_feed_ += feed_req;
408  RecordEnrichment_(feed_req, swu_req);
409 
410  LOG(cyclus::LEV_INFO5, "EnrFac") << prototype()
411  << " has performed an enrichment: ";
412  LOG(cyclus::LEV_INFO5, "EnrFac") << " * Feed Qty: " << feed_req;
413  LOG(cyclus::LEV_INFO5, "EnrFac") << " * Feed Assay: "
414  << assays.Feed() * 100;
415  LOG(cyclus::LEV_INFO5, "EnrFac") << " * Product Qty: " << qty;
416  LOG(cyclus::LEV_INFO5, "EnrFac") << " * Product Assay: "
417  << assays.Product() * 100;
418  LOG(cyclus::LEV_INFO5, "EnrFac") << " * Tails Qty: "
419  << TailsQty(qty, assays);
420  LOG(cyclus::LEV_INFO5, "EnrFac") << " * Tails Assay: "
421  << assays.Tails() * 100;
422  LOG(cyclus::LEV_INFO5, "EnrFac") << " * SWU: " << swu_req;
423  LOG(cyclus::LEV_INFO5, "EnrFac") << " * Current SWU capacity: "
424  << current_swu_capacity;
425 
426  return response;
427 }
