483 // Returns the weight of c using 1 group cross sections of type spectrum
484 // which must be one of:
485 //
486 // * thermal
487 // * thermal_maxwell_ave
488 // * fission_spectrum_ave
489 // * resonance_integral
490 // * fourteen_MeV
491 //
492 // The weight is calculated as "(nu*sigma_f - sigma_a) * N". Since weights
493 // are computed based on nuclide atom fractions, corresponding computed
494 // material/mixing fractions will also be atom-based naturally and will need
495 // to be converted to mass-based for actual material object mixing.
496 double CosiWeight(cyclus::Composition::Ptr c, const std::string& spectrum) {
497  cyclus::CompMap cm = c->atom();
498  cyclus::compmath::Normalize(&cm);
499 
500  if (spectrum == "thermal") {
501  double nu_pu239 = 2.85;
502  double nu_u233 = 2.5;
503  double nu_u235 = 2.43;
504  double nu_u238 = 0;
505  double nu_pu241 = nu_pu239;
506 
507  static std::map<int, double> absorb_xs;
508  static std::map<int, double> fiss_xs;
509  static double p_u238 = 0;
510  static double p_pu239 = 0;
511  if (p_u238 == 0) {
512  double fiss_u238 = simple_xs(922380000, "fission", "thermal");
513  double absorb_u238 = simple_xs(922380000, "absorption", "thermal");
514  p_u238 = nu_u238 * fiss_u238 - absorb_u238;
515 
516  double fiss_pu239 = simple_xs(942390000, "fission", "thermal");
517  double absorb_pu239 = simple_xs(942390000, "absorption", "thermal");
518  p_pu239 = nu_pu239 * fiss_pu239 - absorb_pu239;
519  }
520 
521  cyclus::CompMap::iterator it;
522  double w = 0;
523  for (it = cm.begin(); it != cm.end(); ++it) {
524  cyclus::Nuc nuc = it->first;
525  double nu = 0;
526  if (nuc == 922350000) {
527  nu = nu_u235;
528  } else if (nuc == 922330000) {
529  nu = nu_u233;
530  } else if (nuc == 942390000) {
531  nu = nu_pu239;
532  } else if (nuc == 942410000) {
533  nu = nu_pu241;
534  }
535 
536  double fiss = 0;
537  double absorb = 0;
538  if (absorb_xs.count(nuc) == 0) {
539  try {
540  fiss = simple_xs(nuc, "fission", "thermal");
541  absorb = simple_xs(nuc, "absorption", "thermal");
542  absorb_xs[nuc] = absorb;
543  fiss_xs[nuc] = fiss;
544  } catch (pyne::InvalidSimpleXS err) {
545  fiss = 0;
546  absorb = 0;
547  }
548  } else {
549  fiss = fiss_xs[nuc];
550  absorb = absorb_xs[nuc];
551  }
552 
553  double p = nu * fiss - absorb;
554  w += it->second * (p - p_u238) / (p_pu239 - p_u238);
555  }

557  } else if (spectrum == "fission_spectrum_ave") {

665 // Convert an atom frac (n1/(n1+n2) to a mass frac (m1/(m1+m2) given
666 // corresponding compositions c1 and c2.
667 double AtomToMassFrac(double atomfrac, Composition::Ptr c1,
668  Composition::Ptr c2) {
669  cyclus::CompMap n1 = c1->atom();
670  cyclus::CompMap n2 = c2->atom();
671  cyclus::compmath::Normalize(&n1, atomfrac);
672  cyclus::compmath::Normalize(&n2, 1 - atomfrac);
673 
674  cyclus::CompMap::iterator it;
675 
676  double mass1 = 0;
677  for (it = n1.begin(); it != n1.end(); ++it) {
678  mass1 += it->second * pyne::atomic_mass(it->first);
679  }
680 
681  double mass2 = 0;
682  for (it = n2.begin(); it != n2.end(); ++it) {
683  mass2 += it->second * pyne::atomic_mass(it->first);
684  }
685 
686  return mass1 / (mass1 + mass2);
687 }

689 double HighFrac(double w_low, double w_target, double w_high, double eps) {
690  if (!ValidWeights(w_low, w_target, w_high)) {
691  throw cyclus::ValueError("low and high weights cannot meet target");
692  } else if (w_low == w_high && w_target == w_low) {
693  return 1;
694  }
695  double f = std::abs((w_target - w_low) / (w_high - w_low));
696  if (1 - f < eps) {
697  return 1;
698  } else if (f < eps) {
699  return 0;
700  }
701  return f;
702 }

704 double LowFrac(double w_low, double w_target, double w_high, double eps) {
705  return 1 - HighFrac(w_low, w_target, w_high, eps);
706 }

711  // w_tgt must be in between w_fill and w_fiss for the equivalence
